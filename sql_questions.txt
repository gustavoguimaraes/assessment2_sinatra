create table wizards(
id integer primary key auto_increment,
name text,
age integer,
);

create table powers (
id integer primary key auto_increment,
name text,
damage integer,
wizard_id integer
);

12.

select *
from wizards
join powers on powers.wizard_id = wizards.id
where powers.name like 'l%';

13.
select wizards.name, sum(powers.damage) as total_damage
from wizards
join powers on powers.wizard_id = wizards.id
where powers.damage = 10;

14.

select wizards.name, sum(powers.damage) as total_damage
from wizards
join powers on powers.wizard_id = wizards.id
where powers.damage = 10
order by damage;
